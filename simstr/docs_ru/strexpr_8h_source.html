137/*
138Вспомогательные шаблоны для определения строковых литералов.
139Используются для того, чтобы в параметрах функций ограничивать типы строго как `const K(&)[N]`
140Если пишем
141template<size_t N>
142void func(const char(&lit)[N]);
143то в такую функцию можно будет передать не константный буфер, что может вызывать ошибку:
144
145// Выделили место под символы
146char buf[100];
147// Как то наполнили буфер, допустим, до половины.
148
149stringa text = buf;
150Тут компилятор приведет buf из типа char[100] в тип const char[100] и вызовет конструктор
151для строкового литерала, в text запишется просто указатель на buf и длина 100.
152
153Поэтому такие параметры объявляем как
154template<typename T, typename K = typename const_lit<T>::symb_type, size_t N = const_lit<T>::Count>
155void func(T&& lit);
156
157Тогда компилятор будет подставлять для T точный тип параметра без попыток привести тип к другому типу,
158и выражение с параметром char[100] - не скомпилируется.
159
160Helper templates for defining string literals.
161They are used to limit types in function parameters strictly as `const K(&)[N]`
162If we write
163template<size_t N>
164void func(const char(&lit)[N]);
165then it will be possible to pass a non-constant buffer to such a function, which may cause an error:
166
167// Allocate space for symbols
168char buf[100];
169// Somehow the buffer was filled, say, to half.
170
171stringa text = buf;
172Here the compiler will convert buf from type char[100] to type const char[100] and call the constructor
173for a string literal, text will simply contain a pointer to buf and length 100.
174
175Therefore, we declare such parameters as
176template<typename T, typename K = typename const_lit<T>::symb_type, size_t N = const_lit<T>::Count>
177void func(T&& lit);
178
179Then the compiler will substitute for T the exact type of the parameter without attempting to cast the type to another type,
180and an expression with the char[100] parameter will not compile.
181*/

574/*
575* Шаблонные классы для создания строковых выражений из нескольких источников.
576* Благодаря компиляторно-шаблонной "магии" позволяют максимально эффективно
577* получать результирующую строку - сначала вычисляется длина результирующей строки,
578* потом один раз выделяется память для результата, после символы помещаются в
579* выделенную память.
580* Для конкатенация двух объектов строковых выражений в один
581
582* Template classes for creating string expressions from multiple sources.
583* Thanks to compiler-template "magic" they allow you to maximize efficiency
584* get the resulting string - first the length of the resulting string is calculated,
585* then memory is allocated once for the result, after which the characters are placed in
586* allocated memory.
587* For concatenating two string expression objects into one.
588*/
589
590template<typename K, typename Allocator, StrExpr A>
591constexpr std::basic_string<K, std::char_traits<K>, Allocator> to_std_string(const A& expr) {
592 std::basic_string<K, std::char_traits<K>, Allocator> res;
593 if (size_t l = expr.length()) {
594 auto fill = [&](K* ptr, size_t size) -> size_t {
595 expr.place((typename A::symb_type*)ptr);
596 return l;
597 };
598 if constexpr (requires { res.resize_and_overwrite(l, fill); }) {
599 res.resize_and_overwrite(l, fill);
600 } else if constexpr (requires{ res._Resize_and_overwrite(l, fill); }) {
601 // Work in MSVC std lib before C++23
602 res._Resize_and_overwrite(l, fill);
603 } else {
604 res.resize(l); // bad, fill by 0 first.
605 expr.place((typename A::symb_type*)res.data());
606 }
607 }
608 return res;
609}

1596constexpr size_t fromInt(K* bufEnd, T val) {
1597 const char* twoDigit =
1598 "0001020304050607080910111213141516171819"
1599 "2021222324252627282930313233343536373839"
1600 "4041424344454647484950515253545556575859"
1601 "6061626364656667686970717273747576777879"
1602 "8081828384858687888990919293949596979899";
1603 if (val) {
1604 need_sign<K, std::is_signed_v<T>, T> store(val);
1605 K* itr = bufEnd;
1606 while (store.val >= 100) {
1607 const char* ptr = twoDigit + (store.val % 100) * 2;
1608 *--itr = static_cast<K>(ptr[1]);
1609 *--itr = static_cast<K>(ptr[0]);
1610 store.val /= 100;
1611 }
1612 if (store.val < 10) {
1613 *--itr = static_cast<K>('0' + store.val);

1615 const char* ptr = twoDigit + store.val * 2;
1616 *--itr = static_cast<K>(ptr[1]);
1617 *--itr = static_cast<K>(ptr[0]);
1618 }
1619 store.after(itr);
1620 return size_t(bufEnd - itr);
1621 }
1622 bufEnd[-1] = '0';

2846 #ifndef HAS_BUILTIN_OVERFLOW
2847 u_type maxMult = 0, maxAdd = 0;
2848 if constexpr (CheckOverflow) {
2849 maxMult = std::numeric_limits<u_type>::max() / Base;

2852 #endif

2857

6171template<typename K, size_t NParams>
6172constexpr size_t parse_pattern_string(str_src<K> pattern, auto&& add_part, auto&& add_param) {
6173 char used_args[NParams] = {0};
6174 const K* first = pattern.begin(), *last = pattern.end(), *start = first;
6175 size_t all_len = 0;
6176
6177 auto find = [](const K* from, const K* last, K s) {
6178 while (from != last) {
6179 if (*from == s) {
6180 break;
6181 }
6182 from++;
6183 }
6184 return from;
6185 };
6186 size_t idx_in_params = 0;
6187
6188 while (first != last) {
6189 const K* open_pos = first;
6190 if (*first != '{') {
6191 open_pos = find(first, last, '{');
6192
6193 for (;;) {
6194 const K* close_pos = find(first, open_pos, '}');
6195 if (close_pos == open_pos) {
6196 unsigned len = open_pos - first;
6197 add_part(first - start, len);
6198 all_len += len;
6199 break;
6200 }
6201 ++close_pos;
6202 if (close_pos == open_pos || *close_pos != '}') {
6203 throw parse_subst_string_error{"unescaped }"};
6204 }
6205 unsigned len = close_pos - first;
6206 add_part(first - start, len);
6207 all_len += len;
6208 first = ++close_pos;
6209 }
6210 if (open_pos == last) {
6211 break;
6212 }
6213 }
6214 if (++open_pos == last) {
6215 throw parse_subst_string_error{"unescaped {"};
6216 }
6217 if (*open_pos == '}') {
6218 if (idx_in_params == -1) {
6219 throw parse_subst_string_error{"already used param ids"};
6220 }
6221 if (idx_in_params == NParams) {
6222 throw parse_subst_string_error{"too many params"};
6223 }
6224 used_args[idx_in_params] = 1;
6225 add_param(idx_in_params++);
6226 first = open_pos + 1;
6227 } else if (*open_pos == '{') {
6228 add_part(open_pos - start, 1);
6229 all_len++;
6230 first = open_pos + 1;
6231 } else {
6232 if (idx_in_params != 0 && idx_in_params != -1) {
6233 throw parse_subst_string_error{"already used non id params"};
6234 }
6235 idx_in_params = -1;
6236 const K* end = find(open_pos, last, '}');
6237 if (end == last) {
6238 throw parse_subst_string_error{"not found }"};
6239 }
6240 auto [p, err, _] = str_src<K>(open_pos, end - open_pos).template to_int<unsigned, true, 10, false, false>();
6241 if (err != IntConvertResult::Success || p < 1 || p > NParams) {
6242 throw parse_subst_string_error{"bad param id"};
6243 }
6244 used_args[--p] = 1;
6245 add_param(p);
6246 first = end + 1;
6247 }
6248 }
6249 for (auto c : used_args) {
6250 if (!c) {
6251 throw parse_subst_string_error{"unused param"};
6252 }
6253 }
6254 return all_len;
6255}
