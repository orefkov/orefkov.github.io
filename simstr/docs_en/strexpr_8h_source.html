138/*
139Вспомогательные шаблоны для определения строковых литералов.
140Используются для того, чтобы в параметрах функций ограничивать типы строго как `const K(&)[N]`
141Если пишем
142template<size_t N>
143void func(const char(&lit)[N]);
144то в такую функцию можно будет передать не константный буфер, что может вызывать ошибку:
145
146// Выделили место под символы
147char buf[100];
148// Как то наполнили буфер, допустим, до половины.
149
150stringa text = buf;
151Тут компилятор приведет buf из типа char[100] в тип const char[100] и вызовет конструктор
152для строкового литерала, в text запишется просто указатель на buf и длина 100.
153
154Поэтому такие параметры объявляем как
155template<typename T, typename K = typename const_lit<T>::symb_type, size_t N = const_lit<T>::Count>
156void func(T&& lit);
157
158Тогда компилятор будет подставлять для T точный тип параметра без попыток привести тип к другому типу,
159и выражение с параметром char[100] - не скомпилируется.
160
161Helper templates for defining string literals.
162They are used to limit types in function parameters strictly as `const K(&)[N]`
163If we write
164template<size_t N>
165void func(const char(&lit)[N]);
166then it will be possible to pass a non-constant buffer to such a function, which may cause an error:
167
168// Allocate space for symbols
169char buf[100];
170// Somehow the buffer was filled, say, to half.
171
172stringa text = buf;
173Here the compiler will convert buf from type char[100] to type const char[100] and call the constructor
174for a string literal, text will simply contain a pointer to buf and length 100.
175
176Therefore, we declare such parameters as
177template<typename T, typename K = typename const_lit<T>::symb_type, size_t N = const_lit<T>::Count>
178void func(T&& lit);
179
180Then the compiler will substitute for T the exact type of the parameter without attempting to cast the type to another type,
181and an expression with the char[100] parameter will not compile.
182*/

580/*
581* Шаблонные классы для создания строковых выражений из нескольких источников.
582* Благодаря компиляторно-шаблонной "магии" позволяют максимально эффективно
583* получать результирующую строку - сначала вычисляется длина результирующей строки,
584* потом один раз выделяется память для результата, после символы помещаются в
585* выделенную память.
586* Для конкатенация двух объектов строковых выражений в один
587
588* Template classes for creating string expressions from multiple sources.
589* Thanks to compiler-template "magic" they allow you to maximize efficiency
590* get the resulting string - first the length of the resulting string is calculated,
591* then memory is allocated once for the result, after which the characters are placed in
592* allocated memory.
593* For concatenating two string expression objects into one.
594*/
595
596template<typename K, typename Allocator, StrExpr A>
597constexpr std::basic_string<K, std::char_traits<K>, Allocator> to_std_string(const A& expr) {
598 std::basic_string<K, std::char_traits<K>, Allocator> res;
599 if (size_t l = expr.length()) {
600 auto fill = [&](K* ptr, size_t size) -> size_t {
601 expr.place((typename A::symb_type*)ptr);
602 return l;
603 };
604 if constexpr (requires { res.resize_and_overwrite(l, fill); }) {
605 res.resize_and_overwrite(l, fill);
606 } else if constexpr (requires{ res._Resize_and_overwrite(l, fill); }) {
607 // Work in MSVC std lib before C++23
608 res._Resize_and_overwrite(l, fill);
609 } else {
610 res.resize(l); // bad, fill by 0 first.
611 expr.place((typename A::symb_type*)res.data());
612 }
613 }
614 return res;
615}

1964/*
1965* Для создания строковых конкатенаций с векторами и списками, сджойненными константным разделителем
1966* K - тип символов строки
1967* T - тип контейнера строк (vector, list)
1968* I - длина разделителя в символах
1969* tail - добавлять разделитель после последнего элемента контейнера.
1970* Если контейнер пустой, разделитель в любом случае не добавляется
1971* skip_empty - пропускать пустые строки без добавления разделителя
1972* To create string concatenations with vectors and lists joined by a constant delimiter
1973* K is the symbols
1974* T - type of string container (vector, list)
1975* I - length of separator in characters
1976* tail - add a separator after the last element of the container.
1977* If the container is empty, the separator is not added anyway
1978* skip_empty - skip empty lines without adding a separator
1979*/
1980template<typename K, typename T, size_t I, bool tail, bool skip_empty>
1981struct expr_join : expr_to_std_string<expr_join<K, T, I, tail, skip_empty>> {
1982 using symb_type = K;
1983 using my_type = expr_join<K, T, I, tail, skip_empty>;
1984
1985 const T& s;
1986 const K* delim;
1987 constexpr expr_join(const T& _s, const K* _delim) : s(_s), delim(_delim){}
1988
1989 constexpr size_t length() const noexcept {
1990 size_t l = 0;
1991 for (const auto& t: s) {
1992 size_t len = t.length();
1993 if (len > 0 || !skip_empty) {
1994 if (I > 0 && l > 0) {
1995 l += I;
1996 }
1997 l += len;
1998 }
1999 }
2000 return l + (tail && I > 0 && (l > 0 || (!skip_empty && s.size() > 0))? I : 0);
2001 }
2002 constexpr K* place(K* ptr) const noexcept {
2003 if (s.empty()) {
2004 return ptr;
2005 }
2006 K* write = ptr;
2007 for (const auto& t: s) {
2008 size_t copyLen = t.length();
2009 if (I > 0 && write != ptr && (copyLen || !skip_empty)) {
2010 ch_traits<K>::copy(write, delim, I);
2011 write += I;
2012 }
2013 ch_traits<K>::copy(write, t.data(), copyLen);
2014 write += copyLen;
2015 }
2016 if (I > 0 && tail && (write != ptr || (!skip_empty && s.size() > 0))) {
2017 ch_traits<K>::copy(write, delim, I);
2018 write += I;
2019 }
2020 return write;
2021 }
2022};

5454template<typename K, size_t NParams>
5455constexpr size_t parse_pattern_string(str_src<K> pattern, const auto& add_part, const auto& add_param) {
5456 char used_args[NParams] = {0};
5457 const K* first = pattern.begin(), *last = pattern.end(), *start = first;
5458 size_t all_len = 0;
5459
5460 auto find = [](const K* from, const K* last, K s) {
5461 while (from != last) {
5462 if (*from == s) {
5463 break;
5464 }
5465 from++;
5466 }
5467 return from;
5468 };
5469 size_t idx_in_data = 0, idx_in_params = 0;
5470
5471 while (first != last) {
5472 bool cont = false;
5473 const K* open_pos = first;
5474 if (*first != '{') {
5475 open_pos = find(first, last, '{');
5476
5477 for (;;) {
5478 const K* close_pos = find(first, open_pos, '}');
5479 if (close_pos == open_pos) {
5480 if (open_pos < last && open_pos[1] == '{') {
5481 unsigned len = open_pos - first + 1;
5482 all_len += len;
5483 add_part(first - start, len);
5484 first = open_pos + 2;
5485 cont = true;
5486 } else if (unsigned len = open_pos - first) {
5487 add_part(first - start, len);
5488 all_len += len;
5489 }
5490 break;
5491 }
5492 ++close_pos;
5493 if (close_pos == open_pos || *close_pos != '}') {
5494 throw parse_subst_string_error{"unescaped }"};
5495 }
5496 unsigned len = close_pos - first;
5497 add_part(first - start, len);
5498 all_len += len;
5499 first = ++close_pos;
5500 }
5501 if (open_pos == last) {
5502 break;
5503 }
5504 }
5505 if (cont) {
5506 continue;
5507 }
5508
5509 if (++open_pos == last) {
5510 throw parse_subst_string_error{"unescaped {"};
5511 }
5512 if (*open_pos == '}') {
5513 if (idx_in_params == -1) {
5514 throw parse_subst_string_error{"already used param ids"};
5515 }
5516 used_args[idx_in_params] = 1;
5517 add_param(idx_in_params++);
5518 first = open_pos + 1;
5519 } else if (*open_pos == '{') {
5520 add_part(open_pos - start, 1);
5521 all_len++;
5522 first = open_pos + 1;
5523 } else {
5524 if (idx_in_params != 0 && idx_in_params != -1) {
5525 throw parse_subst_string_error{"already used non id params"};
5526 }
5527 idx_in_params = -1;
5528 const K* end = find(open_pos, last, '}');
5529 if (end == last) {
5530 throw parse_subst_string_error{"not found }"};
5531 }
5532 auto [p, err, _] = str_src<K>(open_pos, end - open_pos).template to_int<unsigned, true, 10, false, false>();
5533 if (err != IntConvertResult::Success || p < 1 || p > NParams) {
5534 throw parse_subst_string_error{"bad param id"};
5535 }
5536 used_args[--p] = 1;
5537 add_param(p);
5538 first = end + 1;
5539 }
5540 }
5541 for (auto c : used_args) {
5542 if (!c) {
5543 throw parse_subst_string_error{"unused param"};
5544 }
5545 }
5546 return all_len;
5547}
