82struct unicode_traits<u8s> {
83 // Эти операции с utf-8 могут изменить длину строки
84 // Поэтому их специализации отличаются
85 // В функцию помимо текста и адреса буфера для записи передается размер буфера
86 // Возвращает длину получающейся строки.
87 // Если получающеюся строка не влезает в отведенный буфер, указатели устанавливаются на последние
88 // обработанные символы, для повторного возобновления работы,
89 // а для оставшихся символов считается нужный размер буфера.
90 // These utf-8 operations can change the length of the string
91 // Therefore their specializations are different
92 // In addition to the text and address of the buffer for writing, the buffer size is passed to the function
93 // Returns the length of the resulting string.
94 // If the resulting string does not fit into the allocated buffer, pointers are set to the last
95 // processed characters to resume work again,
96 // and for the remaining characters the required buffer size is calculated.
97 static SIMSTR_API size_t upper(const u8s*& src, size_t lenStr, u8s*& dest, size_t lenBuf);
98 static SIMSTR_API size_t lower(const u8s*& src, size_t len, u8s*& dest, size_t lenBuf);

284 size_t len = _len();
285 const K* ptr = _str();
286 if constexpr (SkipWS) {
287 while (len && uns_type(*ptr) <= ' ') {
288 len--;
289 ptr++;
290 }
291 }
292 if constexpr (AllowPlus) {
293 if (len && *ptr == K('+')) {
294 ptr++;
295 len--;
296 }
297 }
298 if (!len) {
299 return {};
300 }
301 #ifdef __linux__
302 if constexpr(sizeof(K) == 1) {
303 double d{};
304 if (std::from_chars(ptr, ptr + len, d).ec == std::errc{}) {
305 return d;
306 }
307 return {};
308 }
309 #endif
310 return impl_to_double(ptr, ptr + len);
311 }

630template<typename K>
631class Splitter {
632 simple_str<K> text_;
633 simple_str<K> delim_;
634
635public:
636 constexpr Splitter(simple_str<K> text, simple_str<K> delim) : text_(text), delim_(delim) {}
641 constexpr bool is_done() const {
642 return text_.length() == str::npos;
643 }
644
650 constexpr simple_str<K> next() {
651 if (!text_.length()) {
652 auto ret = text_;
653 text_.str++;
654 text_.len--;
655 return ret;
656 } else if (text_.length() == str::npos) {
657 return {nullptr, 0};
658 }
659 size_t pos = text_.find(delim_), next = 0;
660 if (pos == str::npos) {
661 pos = text_.length();
662 next = pos + 1;
663 } else {
664 next = pos + delim_.length();
665 }
666 simple_str<K> result{text_.str, pos};
667 text_.str += next;
668 text_.len -= next;
669 return result;
670 }
671};
