138/*
139Вспомогательные шаблоны для определения строковых литералов.
140Используются для того, чтобы в параметрах функций ограничивать типы строго как `const K(&)[N]`
141Если пишем
142template<size_t N>
143void func(const char(&lit)[N]);
144то в такую функцию можно будет передать не константный буфер, что может вызывать ошибку:
145
146// Выделили место под символы
147char buf[100];
148// Как то наполнили буфер, допустим, до половины.
149
150stringa text = buf;
151Тут компилятор приведет buf из типа char[100] в тип const char[100] и вызовет конструктор
152для строкового литерала, в text запишется просто указатель на buf и длина 100.
153
154Поэтому такие параметры объявляем как
155template<typename T, typename K = typename const_lit<T>::symb_type, size_t N = const_lit<T>::Count>
156void func(T&& lit);
157
158Тогда компилятор будет подставлять для T точный тип параметра без попыток привести тип к другому типу,
159и выражение с параметром char[100] - не скомпилируется.
160
161Helper templates for defining string literals.
162They are used to limit types in function parameters strictly as `const K(&)[N]`
163If we write
164template<size_t N>
165void func(const char(&lit)[N]);
166then it will be possible to pass a non-constant buffer to such a function, which may cause an error:
167
168// Allocate space for symbols
169char buf[100];
170// Somehow the buffer was filled, say, to half.
171
172stringa text = buf;
173Here the compiler will convert buf from type char[100] to type const char[100] and call the constructor
174for a string literal, text will simply contain a pointer to buf and length 100.
175
176Therefore, we declare such parameters as
177template<typename T, typename K = typename const_lit<T>::symb_type, size_t N = const_lit<T>::Count>
178void func(T&& lit);
179
180Then the compiler will substitute for T the exact type of the parameter without attempting to cast the type to another type,
181and an expression with the char[100] parameter will not compile.
182*/

577/*
578* Шаблонные классы для создания строковых выражений из нескольких источников.
579* Благодаря компиляторно-шаблонной "магии" позволяют максимально эффективно
580* получать результирующую строку - сначала вычисляется длина результирующей строки,
581* потом один раз выделяется память для результата, после символы помещаются в
582* выделенную память.
583* Для конкатенация двух объектов строковых выражений в один
584
585* Template classes for creating string expressions from multiple sources.
586* Thanks to compiler-template "magic" they allow you to maximize efficiency
587* get the resulting string - first the length of the resulting string is calculated,
588* then memory is allocated once for the result, after which the characters are placed in
589* allocated memory.
590* For concatenating two string expression objects into one.
591*/
592
593template<typename K, typename Allocator, StrExpr A>
594constexpr std::basic_string<K, std::char_traits<K>, Allocator> to_std_string(const A& expr) {
595 std::basic_string<K, std::char_traits<K>, Allocator> res;
596 if (size_t l = expr.length()) {
597 auto fill = [&](K* ptr, size_t size) -> size_t {
598 expr.place((typename A::symb_type*)ptr);
599 return l;
600 };
601 if constexpr (requires { res.resize_and_overwrite(l, fill); }) {
602 res.resize_and_overwrite(l, fill);
603 } else if constexpr (requires{ res._Resize_and_overwrite(l, fill); }) {
604 // Work in MSVC std lib before C++23
605 res._Resize_and_overwrite(l, fill);
606 } else {
607 res.resize(l); // bad, fill by 0 first.
608 expr.place((typename A::symb_type*)res.data());
609 }
610 }
611 return res;
612}

1601template<typename K, typename T>
1602constexpr size_t fromInt(K* bufEnd, T val) {
1603 const char* twoDigit =
1604 "0001020304050607080910111213141516171819"
1605 "2021222324252627282930313233343536373839"
1606 "4041424344454647484950515253545556575859"
1607 "6061626364656667686970717273747576777879"
1608 "8081828384858687888990919293949596979899";
1609 if (val) {
1610 need_sign<K, std::is_signed_v<T>, T> store(val);
1611 K* itr = bufEnd;
1612 while (store.val >= 100) {
1613 const char* ptr = twoDigit + (store.val % 100) * 2;
1614 *--itr = static_cast<K>(ptr[1]);
1615 *--itr = static_cast<K>(ptr[0]);
1616 store.val /= 100;
1617 }
1618 if (store.val < 10) {
1619 *--itr = static_cast<K>('0' + store.val);
1620 } else {
1621 const char* ptr = twoDigit + store.val * 2;
1622 *--itr = static_cast<K>(ptr[1]);
1623 *--itr = static_cast<K>(ptr[0]);
1624 }
1625 store.after(itr);
1626 return size_t(bufEnd - itr);
1627 }
1628 bufEnd[-1] = '0';
1629 return 1;
1630}

5937template<typename K, size_t NParams>
5938constexpr size_t parse_pattern_string(str_src<K> pattern, auto&& add_part, auto&& add_param) {
5939 char used_args[NParams] = {0};
5940 const K* first = pattern.begin(), *last = pattern.end(), *start = first;
5941 size_t all_len = 0;
5942
5943 auto find = [](const K* from, const K* last, K s) {
5944 while (from != last) {
5945 if (*from == s) {
5946 break;
5947 }
5948 from++;
5949 }
5950 return from;
5951 };
5952 size_t idx_in_params = 0;
5953
5954 while (first != last) {
5955 const K* open_pos = first;
5956 if (*first != '{') {
5957 open_pos = find(first, last, '{');
5958
5959 for (;;) {
5960 const K* close_pos = find(first, open_pos, '}');
5961 if (close_pos == open_pos) {
5962 unsigned len = open_pos - first;
5963 add_part(first - start, len);
5964 all_len += len;
5965 break;
5966 }
5967 ++close_pos;
5968 if (close_pos == open_pos || *close_pos != '}') {
5969 throw parse_subst_string_error{"unescaped }"};
5970 }
5971 unsigned len = close_pos - first;
5972 add_part(first - start, len);
5973 all_len += len;
5974 first = ++close_pos;
5975 }
5976 if (open_pos == last) {
5977 break;
5978 }
5979 }
5980 if (++open_pos == last) {
5981 throw parse_subst_string_error{"unescaped {"};
5982 }
5983 if (*open_pos == '}') {
5984 if (idx_in_params == -1) {
5985 throw parse_subst_string_error{"already used param ids"};
5986 }
5987 if (idx_in_params == NParams) {
5988 throw parse_subst_string_error{"too many params"};
5989 }
5990 used_args[idx_in_params] = 1;
5991 add_param(idx_in_params++);
5992 first = open_pos + 1;
5993 } else if (*open_pos == '{') {
5994 add_part(open_pos - start, 1);
5995 all_len++;
5996 first = open_pos + 1;
5997 } else {
5998 if (idx_in_params != 0 && idx_in_params != -1) {
5999 throw parse_subst_string_error{"already used non id params"};
6000 }
6001 idx_in_params = -1;
6002 const K* end = find(open_pos, last, '}');
6003 if (end == last) {
6004 throw parse_subst_string_error{"not found }"};
6005 }
6006 auto [p, err, _] = str_src<K>(open_pos, end - open_pos).template to_int<unsigned, true, 10, false, false>();
6007 if (err != IntConvertResult::Success || p < 1 || p > NParams) {
6008 throw parse_subst_string_error{"bad param id"};
6009 }
6010 used_args[--p] = 1;
6011 add_param(p);
6012 first = end + 1;
6013 }
6014 }
6015 for (auto c : used_args) {
6016 if (!c) {
6017 throw parse_subst_string_error{"unused param"};
6018 }
6019 }
6020 return all_len;
6021}
