91/*
92Вспомогательные шаблоны для определения строковых литералов.
93Используются для того, чтобы в параметрах функций ограничивать типы строго как `const K(&)[N]`
94Если пишем
95template<size_t N>
96void func(const char(&lit)[N]);
97то в такую функцию можно будет передать не константный буфер, что может вызывать ошибку:
98
99// Выделили место под символы
100char buf[100];
101// Как то наполнили буфер, допустим, до половины.
102
103stringa text = buf;
104Тут компилятор приведет buf из типа char[100] в тип const char[100] и вызовет конструктор
105для строкового литерала, в text запишется просто указатель на buf и длина 100.
106
107Поэтому такие параметры объявляем как
108template<typename T, typename K = typename const_lit<T>::symb_type, size_t N = const_lit<T>::Count>
109void func(T&& lit);
110
111Тогда компилятор будет подставлять для T точный тип параметра без попыток привести тип к другому типу,
112и выражение с параметром char[100] - не скомпилируется.
113
114Helper templates for defining string literals.
115They are used to limit types in function parameters strictly as `const K(&)[N]`
116If we write
117template<size_t N>
118void func(const char(&lit)[N]);
119then it will be possible to pass a non-constant buffer to such a function, which may cause an error:
120
121// Allocate space for symbols
122char buf[100];
123// Somehow the buffer was filled, say, to half.
124
125stringa text = buf;
126Here the compiler will convert buf from type char[100] to type const char[100] and call the constructor
127for a string literal, text will simply contain a pointer to buf and length 100.
128
129Therefore, we declare such parameters as
130template<typename T, typename K = typename const_lit<T>::symb_type, size_t N = const_lit<T>::Count>
131void func(T&& lit);
132
133Then the compiler will substitute for T the exact type of the parameter without attempting to cast the type to another type,
134and an expression with the char[100] parameter will not compile.
135*/

387/*
388* Шаблонные классы для создания строковых выражений из нескольких источников.
389* Благодаря компиляторно-шаблонной "магии" позволяют максимально эффективно
390* получать результирующую строку - сначала вычисляется длина результирующей строки,
391* потом один раз выделяется память для результата, после символы помещаются в
392* выделенную память.
393* Для конкатенация двух объектов строковых выражений в один
394
395* Template classes for creating string expressions from multiple sources.
396* Thanks to compiler-template "magic" they allow you to maximize efficiency
397* get the resulting string - first the length of the resulting string is calculated,
398* then memory is allocated once for the result, after which the characters are placed in
399* allocated memory.
400* For concatenating two string expression objects into one.
401*/

403template<typename K, typename Allocator, StrExpr A>
404constexpr std::basic_string<K, std::char_traits<K>, Allocator> to_std_string(const A& expr) {
405 std::basic_string<K, std::char_traits<K>, Allocator> res;
406 size_t l = expr.length();
407 auto fill = [&](K* ptr, size_t size) -> size_t {
408 expr.place((typename A::symb_type*)ptr);
409 return size;
410 };
411 if (l) {
412 if constexpr (requires{res.resize_and_overwrite(l, fill);}) {
413 res.resize_and_overwrite(l, fill);
414 } else if constexpr (requires{res._Resize_and_overwrite(l, fill);}) {
415 // Work in MSVC std lib before C++23
416 res._Resize_and_overwrite(l, fill);
417 } else {
418 res.resize(l); // bad, fill by 0 first.
419 fill((typename A::symb_type*)res.data(), l);
420 }
421 }
422 return res;
423}

1386constexpr size_t fromInt(K* bufEnd, T val) {
1387 const char* twoDigit =
1388 "0001020304050607080910111213141516171819"
1389 "2021222324252627282930313233343536373839"
1390 "4041424344454647484950515253545556575859"
1391 "6061626364656667686970717273747576777879"
1392 "8081828384858687888990919293949596979899";
1393 if (val) {
1394 need_sign<K, std::is_signed_v<T>, T> sign(val);
1395 K* itr = bufEnd;
1396 // Когда у нас минимальное отрицательное число, оно не меняется и остается меньше нуля
1397 // When we have a minimum negative number, it does not change and remains less than zero
1398 if constexpr (std::is_signed_v<T>) {
1399 if (val < 0) {
1400 // Возьмем две последние цифры
1401 // Take the last two digits
1402 const char* ptr = twoDigit - (val % 100) * 2;
1403 *--itr = static_cast<K>(ptr[1]);
1404 *--itr = static_cast<K>(ptr[0]);
1405 val /= 100;
1406 val = -val;
1407 }
1408 }
1409 while (val >= 100) {
1410 const char* ptr = twoDigit + (val % 100) * 2;
1411 *--itr = static_cast<K>(ptr[1]);
1412 *--itr = static_cast<K>(ptr[0]);
1413 val /= 100;
1414 }
1415 if (val < 10) {
1416 *--itr = static_cast<K>('0' + val);
1417 } else {
1418 const char* ptr = twoDigit + val * 2;
1419 *--itr = static_cast<K>(ptr[1]);
1420 *--itr = static_cast<K>(ptr[0]);
1421 }
1422 sign.after(itr);
1423 return size_t(bufEnd - itr);
1424 }
1425 bufEnd[-1] = '0';

1817* skip_empty - пропускать пустые строки без добавления разделителя

1819* K is the symbols
1820* T - type of string container (vector, list)
1821* I - length of separator in characters
1822* tail - add a separator after the last element of the container.
1823* If the container is empty, the separator is not added anyway
1824* skip_empty - skip empty lines without adding a separator

1836 size_t l = 0;
1837 for (const auto& t: s) {
1838 size_t len = t.length();
1839 if (len > 0 || !skip_empty) {
1840 if (I > 0 && l > 0) {
1841 l += I;
1842 }
1843 l += len;
1844 }
1845 }
1846 return l + (tail && I > 0 && (l > 0 || (!skip_empty && s.size() > 0))? I : 0);

1850 return ptr;
1851 }

1855 if (I > 0 && write != ptr && (copyLen || !skip_empty)) {

4232 if (fnd == str::npos) {
4233 return l;
4234 }
4235 l += replaces_[num].len - 1;

4237 search_results_.reserve((l >> 4) + 8);
4238 search_results_.emplace_back(fnd, num);
4239 for (size_t start = fnd + 1;;) {
4240 auto [fnd, idx] = find_first_of(source_.str, source_.len, start);
4241 if (fnd == str::npos) {
4242 break;
4243 }
4244 search_results_.emplace_back(fnd, idx);
4245 start = fnd + 1;
4246 l += replaces_[idx].len - 1;
4247 }
4248 } else {

4250 search_results_.replaces_[0] = {fnd, num};
4251 search_results_.count_++;
4252 for (size_t start = fnd + 1;;) {
4253 auto [found, idx] = find_first_of(source_.str, source_.len, start);
4254 if (found == str::npos) {
4255 break;
4256 }
4257 if (search_results_.count_ < max_store) {
4258 search_results_.replaces_[search_results_.count_] = {found, idx};
4259 }

4261 search_results_.count_++;

4263 }
4264 }
4265 return l;

4271 for (const auto& [pos, num] : search_results_) {
4272 size_t delta = pos - start;
4273 ch_traits<K>::copy(ptr, text + start, delta);
4274 ptr += delta;
4275 ptr = replaces_[num].place(ptr);
4276 start = pos + 1;
4277 }
4278 } else {
4279 const size_t max_store = std::size(search_results_.replaces_);
4280 size_t founded = search_results_.count_;

4282 const auto [pos, num] = search_results_.replaces_[idx];
4283 size_t delta = pos - start;
4284 ch_traits<K>::copy(ptr, text + start, delta);
4285 ptr += delta;
4286 ptr = replaces_[num].place(ptr);
4287 start = pos + 1;
4288 }
4289 if (founded > max_store) {
4290 founded -= max_store;
4291 while (founded--) {
4292 auto [fnd, idx] = find_first_of(source_.str, source_.len, start);
4293 size_t delta = fnd - start;
4294 ch_traits<K>::copy(ptr, text + start, delta);
4295 ptr += delta;
4296 ptr = replaces_[idx].place(ptr);
4297 start = fnd + 1;
4298 }
4299 }
4300 }

4322 bit_mask_[32 + (s >> 3)] |= 1 << (s & 7);
4323 } else {

4356 }
4357 } else {
4358 if (is_in_mask(s)) {
4359 idx = index_of<0>(s);

4361 }
4362 }
4363 }
4364 }

4373 }
4374 offset++;
4375 }
4376 } else {

4378 if (size_t idx = index_of<0>(text[offset]); idx != -1) {
4379 return {offset, idx};
4380 }
4381 offset++;
4382 }
4383 }
4384 return {-1, -1};
